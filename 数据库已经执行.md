CREATE OR REPLACE FUNCTION update_favorite_count()
RETURNS TRIGGER AS $$
BEGIN
    IF TG_OP = 'INSERT' THEN
        IF NEW.is_favorited = TRUE THEN
            UPDATE profiles 
            SET favorite_count = favorite_count + 1 
            WHERE id = NEW.user_id;
        END IF;
        RETURN NEW;
    ELSIF TG_OP = 'UPDATE' THEN
        IF OLD.is_favorited = FALSE AND NEW.is_favorited = TRUE THEN
            UPDATE profiles 
            SET favorite_count = favorite_count + 1 
            WHERE id = NEW.user_id;
        ELSIF OLD.is_favorited = TRUE AND NEW.is_favorited = FALSE THEN
            UPDATE profiles 
            SET favorite_count = favorite_count - 1 
            WHERE id = NEW.user_id;
        END IF;
        RETURN NEW;
    ELSIF TG_OP = 'DELETE' THEN
        IF OLD.is_favorited = TRUE THEN
            UPDATE profiles 
            SET favorite_count = favorite_count - 1 
            WHERE id = OLD.user_id;
        END IF;
        RETURN OLD;
    END IF;
    RETURN NULL;
END;
$$ LANGUAGE plpgsql;

-- 13. 创建实用函数
-- 检查用户使用限制的函数
CREATE OR REPLACE FUNCTION check_user_usage_limit(user_uuid UUID, operation_type TEXT)
RETURNS BOOLEAN AS $$
DECLARE
    user_profile profiles%ROWTYPE;
    max_limit INT;
    current_count INT;
BEGIN
    -- 获取用户信息
    SELECT * INTO user_profile FROM profiles WHERE id = user_uuid;
    
    IF NOT FOUND THEN
        RETURN FALSE;
    END IF;
    
    -- 检查是否需要重置计数
    IF user_profile.usage_last_reset != CURRENT_DATE THEN
        UPDATE profiles 
        SET generation_count = 0, rewrite_count = 0, usage_last_reset = CURRENT_DATE
        WHERE id = user_uuid;
        
        user_profile.generation_count := 0;
        user_profile.rewrite_count := 0;
    END IF;
    
    -- 根据用户类型和操作类型确定限制
    IF user_profile.status = 'free' THEN
        IF operation_type = 'generation' THEN
            max_limit := 1;
            current_count := user_profile.generation_count;
        ELSIF operation_type = 'rewrite' THEN
            max_limit := 1;
            current_count := user_profile.rewrite_count;
        END IF;
    ELSE
        IF operation_type = 'generation' THEN
            max_limit := 30;
            current_count := user_profile.generation_count;
        ELSIF operation_type = 'rewrite' THEN
            max_limit := 30;
            current_count := user_profile.rewrite_count;
        END IF;
    END IF;
    
    RETURN current_count < max_limit;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

-- 检查收藏限制的函数
CREATE OR REPLACE FUNCTION check_favorite_limit(user_uuid UUID)
RETURNS BOOLEAN AS $$
DECLARE
    user_profile profiles%ROWTYPE;
    max_favorites INT;
BEGIN
    SELECT * INTO user_profile FROM profiles WHERE id = user_uuid;
    
    IF NOT FOUND THEN
        RETURN FALSE;
    END IF;
    
    IF user_profile.status = 'free' THEN
        max_favorites := 3;
    ELSE
        max_favorites := 100;
    END IF;
    
    RETURN user_profile.favorite_count < max_favorites;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

-- 2) 将处理函数设为 SECURITY DEFINER（确保定时任务/系统可处理，不受 RLS 影响）
create or replace function public.process_paddle_webhooks(batch_size int default 50)
returns void
language plpgsql
security definer
set search_path = public
as $$
declare
  rec record;
  customer_email text;
  customer_id text;
  price_id text;
begin
  for rec in
    select * from paddle_webhook_queue
    where status='pending'
    order by received_at asc
    limit batch_size
    for update skip locked
  loop
    begin
      if rec.event_type in ('subscription.created','subscription.activated') then
        customer_email := rec.payload->'data'->'customer'->>'email';
        customer_id    := rec.payload->'data'->'customer'->>'id';
        price_id       := rec.payload->'data'->'subscription'->'items'->0->'price'->>'id';
        update profiles
          set status='active',
              paddle_customer_id=customer_id,
              active_price_id=price_id,
              updated_at=now()
        where email=customer_email;
      elsif rec.event_type in ('subscription.updated') then
        price_id := rec.payload->'data'->'subscription'->'items'->0->'price'->>'id';
        update profiles
          set active_price_id=price_id,
              updated_at=now()
        where paddle_customer_id=rec.payload->'data'->'customer'->>'id';
      elsif rec.event_type in ('subscription.canceled','subscription.cancelled') then
        update profiles
          set status='canceled',
              updated_at=now()
        where paddle_customer_id=rec.payload->'data'->'customer'->>'id';
      end if;

      update paddle_webhook_queue
        set status='done', processed_at=now(), error=null
      where id=rec.id;
    exception when others then
      update paddle_webhook_queue
        set status='error', attempts=attempts+1, error=left(sqlerrm, 400)
      where id=rec.id;
    end;
  end loop;
end $$;

CREATE OR REPLACE FUNCTION update_favorite_count_optimized()
RETURNS TRIGGER AS $$
BEGIN
  -- 只在真正需要更新时才执行
  IF TG_OP = 'INSERT' AND NEW.is_favorited = TRUE THEN
    -- 使用单次更新，减少数据库操作
    UPDATE profiles 
    SET favorite_count = favorite_count + 1,
        updated_at = NOW()
    WHERE id = NEW.user_id;
    RETURN NEW;
    
  ELSIF TG_OP = 'UPDATE' THEN
    -- 只在收藏状态真正改变时才更新
    IF OLD.is_favorited = FALSE AND NEW.is_favorited = TRUE THEN
      UPDATE profiles 
      SET favorite_count = favorite_count + 1,
          updated_at = NOW()
      WHERE id = NEW.user_id;
    ELSIF OLD.is_favorited = TRUE AND NEW.is_favorited = FALSE THEN
      UPDATE profiles 
      SET favorite_count = GREATEST(0, favorite_count - 1),
          updated_at = NOW()
      WHERE id = NEW.user_id;
    END IF;
    RETURN NEW;
    
  ELSIF TG_OP = 'DELETE' AND OLD.is_favorited = TRUE THEN
    UPDATE profiles 
    SET favorite_count = GREATEST(0, favorite_count - 1),
        updated_at = NOW()
    WHERE id = OLD.user_id;
    RETURN OLD;
  END IF;
  
  RETURN COALESCE(NEW, OLD);
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

-- 4. 创建检查用户试用期状态的函数
CREATE OR REPLACE FUNCTION is_user_in_trial_period(user_uuid UUID)
RETURNS BOOLEAN AS $$
DECLARE
    user_profile profiles%ROWTYPE;
BEGIN
    -- 获取用户信息
    SELECT * INTO user_profile FROM profiles WHERE id = user_uuid;
    
    IF NOT FOUND THEN
        RETURN FALSE;
    END IF;
    
    -- 检查是否在试用期内
    IF user_profile.trial_start_date IS NOT NULL 
       AND user_profile.trial_end_date IS NOT NULL 
       AND NOW() >= user_profile.trial_start_date 
       AND NOW() <= user_profile.trial_end_date THEN
        RETURN TRUE;
    END IF;
    
    RETURN FALSE;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

-- 5. 创建检查用户是否有试用权限的函数
CREATE OR REPLACE FUNCTION can_user_use_trial(user_uuid UUID)
RETURNS BOOLEAN AS $$
DECLARE
    user_profile profiles%ROWTYPE;
BEGIN
    -- 获取用户信息
    SELECT * INTO user_profile FROM profiles WHERE id = user_uuid;
    
    IF NOT FOUND THEN
        RETURN FALSE;
    END IF;
    
    -- 如果已经是付费用户，不需要试用
    IF user_profile.status = 'active' THEN
        RETURN FALSE;
    END IF;
    
    -- 如果已经使用过试用期，不能再使用
    IF user_profile.is_trial_used = TRUE THEN
        RETURN FALSE;
    END IF;
    
    -- 如果正在试用期内，可以继续使用
    IF is_user_in_trial_period(user_uuid) THEN
        RETURN TRUE;
    END IF;
    
    -- 如果从未使用过试用期，可以使用
    IF user_profile.trial_start_date IS NULL AND user_profile.trial_end_date IS NULL THEN
        RETURN TRUE;
    END IF;
    
    RETURN FALSE;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

-- 6. 创建激活用户试用期的函数
CREATE OR REPLACE FUNCTION activate_user_trial(user_uuid UUID)
RETURNS BOOLEAN AS $$
DECLARE
    trial_start TIMESTAMPTZ;
    trial_end TIMESTAMPTZ;
BEGIN
    -- 检查用户是否有试用权限
    IF NOT can_user_use_trial(user_uuid) THEN
        RETURN FALSE;
    END IF;
    
    -- 设置试用期时间（3天）
    trial_start := NOW();
    trial_end := NOW() + INTERVAL '3 days';
    
    -- 更新用户试用期信息
    UPDATE profiles 
    SET 
        trial_start_date = trial_start,
        trial_end_date = trial_end,
        is_trial_used = TRUE,
        updated_at = NOW()
    WHERE id = user_uuid;
    
    RETURN TRUE;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

-- 7. 创建检查用户使用限制的函数（支持试用期）
CREATE OR REPLACE FUNCTION check_user_usage_limit_with_trial(user_uuid UUID, operation_type TEXT)
RETURNS BOOLEAN AS $$
DECLARE
    user_profile profiles%ROWTYPE;
    max_limit INT;
    current_count INT;
    is_in_trial BOOLEAN;
BEGIN
    -- 获取用户信息
    SELECT * INTO user_profile FROM profiles WHERE id = user_uuid;
    
    IF NOT FOUND THEN
        RETURN FALSE;
    END IF;
    
    -- 检查是否在试用期内
    is_in_trial := is_user_in_trial_period(user_uuid);
    
    -- 检查是否需要重置计数
    IF user_profile.usage_last_reset != CURRENT_DATE THEN
        UPDATE profiles 
        SET generation_count = 0, rewrite_count = 0, usage_last_reset = CURRENT_DATE
        WHERE id = user_uuid;
        
        user_profile.generation_count := 0;
        user_profile.rewrite_count := 0;
    END IF;
    
    -- 根据用户类型和操作类型确定限制
    IF user_profile.status = 'active' OR is_in_trial THEN
        -- 付费用户或试用期用户享有付费用户权益
        IF operation_type = 'generation' THEN
            max_limit := 30;
            current_count := user_profile.generation_count;
        ELSIF operation_type = 'rewrite' THEN
            max_limit := 30;
            current_count := user_profile.rewrite_count;
        END IF;
    ELSE
        -- 免费用户
        IF operation_type = 'generation' THEN
            max_limit := 1;
            current_count := user_profile.generation_count;
        ELSIF operation_type = 'rewrite' THEN
            max_limit := 1;
            current_count := user_profile.rewrite_count;
        END IF;
    END IF;
    
    RETURN current_count < max_limit;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

-- 8. 创建检查收藏限制的函数（支持试用期）
CREATE OR REPLACE FUNCTION check_favorite_limit_with_trial(user_uuid UUID)
RETURNS BOOLEAN AS $$
DECLARE
    user_profile profiles%ROWTYPE;
    max_favorites INT;
    is_in_trial BOOLEAN;
BEGIN
    SELECT * INTO user_profile FROM profiles WHERE id = user_uuid;
    
    IF NOT FOUND THEN
        RETURN FALSE;
    END IF;
    
    -- 检查是否在试用期内
    is_in_trial := is_user_in_trial_period(user_uuid);
    
    IF user_profile.status = 'active' OR is_in_trial THEN
        max_favorites := 100;
    ELSE
        max_favorites := 3;
    END IF;
    
    RETURN user_profile.favorite_count < max_favorites;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

CREATE OR REPLACE FUNCTION update_favorite_count_smart()
RETURNS TRIGGER AS $$
BEGIN
  IF TG_OP = 'INSERT' AND NEW.is_favorited = TRUE THEN
    UPDATE profiles 
    SET favorite_count = favorite_count + 1,
        updated_at = NOW()
    WHERE id = NEW.user_id;
  ELSIF TG_OP = 'UPDATE' AND OLD.is_favorited != NEW.is_favorited THEN
    UPDATE profiles 
    SET favorite_count = favorite_count + CASE WHEN NEW.is_favorited THEN 1 ELSE -1 END,
        updated_at = NOW()
    WHERE id = NEW.user_id;
  ELSIF TG_OP = 'DELETE' AND OLD.is_favorited = TRUE THEN
    UPDATE profiles 
    SET favorite_count = GREATEST(0, favorite_count - 1),
        updated_at = NOW()
    WHERE id = OLD.user_id;
  END IF;
  RETURN COALESCE(NEW, OLD);
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;
